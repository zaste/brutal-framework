@brutal/animation

Ultra-lightweight animation library with GPU-accelerated transforms

Transform

animate(element, { x: 100, y: -50, rotate: 180, scale: 1.5 }, { duration: 600, easing: 'easeInOut' });

Fade In/Out

animate(element, { opacity: visible ? 0 : 1 }, { duration: 300, onComplete: () => { visible = !visible; } });

Easing Functions

// Linear animate(el1, { x: 200 }, { easing: 'linear' }); // Ease In animate(el2, { x: 200 }, { easing: 'easeIn' }); // Ease Out animate(el3, { x: 200 }, { easing: 'easeOut' }); // Ease In Out animate(el4, { x: 200 }, { easing: 'easeInOut' });

Timeline Sequence

timeline([ [box1, { y: -60, scale: 1.5 }, { duration: 200 }], [box2, { y: -60, scale: 1.5 }, { duration: 200 }], [box3, { y: -60, scale: 1.5 }, { duration: 200 }], [box4, { y: -60, scale: 1.5 }, { duration: 200 }] ]);

Chained Animations

animate(box, { x: 100 }, { duration: 300 }) .promise.then(() => animate(box, { y: -50 }, { duration: 300 }).promise ).then(() => animate(box, { rotate: 360, scale: 0 }, { duration: 500 }).promise );

Interruptible

// Store reference currentAnim = animate(element, { x: 150, rotate: 720 }, { duration: 2000 }); // Interrupt later currentAnim.stop();
