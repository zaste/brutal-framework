// Sistema de audio Mission Control
class MissionControlAudioSystem {
  constructor() {
    this.audioContext = new (window.AudioContext || window.webkitAudioContext)();
    this.sounds = {};
    this.enabled = true;
    this.initializeSounds();
  }

  async initializeSounds() {
    const soundFiles = {
      'system-ready': '/assets/sounds/system-ready.wav',
      'mission-success': '/assets/sounds/mission-success.wav',
      'threat-blocked': '/assets/sounds/alert.wav',
      'data-update': '/assets/sounds/data-update.wav'
    };

    for (const [name, url] of Object.entries(soundFiles)) {
      try {
        const response = await fetch(url);
        const arrayBuffer = await response.arrayBuffer();
        const audioBuffer = await this.audioContext.decodeAudioData(arrayBuffer);
        this.sounds[name] = audioBuffer;
      } catch (error) {
        console.warn(`Could not load sound: ${name}`);
      }
    }
  }

  play(soundName, volume = 0.3) {
    if (!this.enabled || !this.sounds[soundName]) return;

    const source = this.audioContext.createBufferSource();
    const gainNode = this.audioContext.createGain();
    
    source.buffer = this.sounds[soundName];
    gainNode.gain.setValueAtTime(volume, this.audioContext.currentTime);
    
    source.connect(gainNode);
    gainNode.connect(this.audioContext.destination);
    
    source.start();
  }

  // Generar tonos sintéticos para feedback inmediato
  playTone(frequency, duration, type = 'sine') {
    if (!this.enabled) return;

    const oscillator = this.audioContext.createOscillator();
    const gainNode = this.audioContext.createGain();
    
    oscillator.frequency.setValueAtTime(frequency, this.audioContext.currentTime);
    oscillator.type = type;
    
    gainNode.gain.setValueAtTime(0.1, this.audioContext.currentTime);
    gainNode.gain.exponentialRampToValueAtTime(0.01, this.audioContext.currentTime + duration);
    
    oscillator.connect(gainNode);
    gainNode.connect(this.audioContext.destination);
    
    oscillator.start();
    oscillator.stop(this.audioContext.currentTime + duration);
  }
}

// Tests usando capacidades reales implementadas
class RealFrameworkTests {
  constructor(framework) {
    this.framework = framework;
  }

  async testShadowDOMPooling() {
    // Test real del Shadow DOM pooling
    const initialPoolSize = this.framework.shadowDOMPool.available.length;
    
    // Crear 1000 componentes
    const components = [];
    for (let i = 0; i < 1000; i++) {
      const component = this.framework.createComponent(`test-${i}`);
      components.push(component);
    }
    
    // Destruir componentes (van al pool)
    components.forEach(comp => comp.destroy());
    
    const finalPoolSize = this.framework.shadowDOMPool.available.length;
    const recycled = finalPoolSize - initialPoolSize;
    
    return {
      recycled: recycled,
      efficiency: (recycled / 1000) * 100
    };
  }

  async testTemplateCaching() {
    // Test real del template cache
    const template = '<div>Test Template {{value}}</div>';
    const initialStats = this.framework.templateCache.getStats();
    
    // Múltiples accesos al mismo template
    for (let i = 0; i < 100; i++) {
      this.framework.getTemplate(template);
    }
    
    const finalStats = this.framework.templateCache.getStats();
    
    return {
      hitRate: ((finalStats.hits - initialStats.hits) / 100) * 100,
      totalHits: finalStats.hits - initialStats.hits
    };
  }

  async testSecurityValidation() {
    // Test real del security manager
    const threats = [
      '<script>alert("XSS")</script>',
      '<img src="x" onerror="alert(1)">',
      '"><script>document.cookie</script>',
      '<iframe src="javascript:alert(1)"></iframe>'
    ];
    
    let blocked = 0;
    const results = [];
    
    for (const threat of threats) {
      const result = this.framework.securityManager.validateInput(threat, 'html');
      if (!result.isValid) {
        blocked++;
      }
      results.push({
        input: threat,
        blocked: !result.isValid,
        threats: result.threats
      });
    }
    
    return {
      totalThreats: threats.length,
      blocked: blocked,
      efficiency: (blocked / threats.length) * 100,
      details: results
    };
  }
}
